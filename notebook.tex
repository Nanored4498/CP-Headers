	\paragraph{Fibonacci}
	$$ F_{n+k} = F_k F_{n+1} + F_{k-1} F_n \qquad GCD(F_m, F_n) = F_{GCD(m, n)} $$
	Use the first equation with $k=n$ or $k=n+1$ for fast computation.
	\paragraph{Number theory}
	$$ x^n \equiv x^{\phi(m) + \left( n \!\!\!\mod \phi(m) \right)} \mod m $$
